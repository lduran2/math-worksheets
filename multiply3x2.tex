\begin{document}

\thispagestyle{empty}

\newcounter{randtwo}

\begin{enumerate}
    \foreach \k in {1,...,108}{
        \item
            \pgfmathparse{random(2)}
            \setcounter{randtwo}{\pgfmathresult}
            \ifnum\therandtwo=1
                Compute:
            \else
                Evaluate:
            \fi
            \(
                \pgfmathparse{int(random(100,999))}
                    \pgfmathresult
                \times
                \pgfmathparse{int(random(10,99))}
                    \pgfmathresult
            \)
    }
\end{enumerate}
